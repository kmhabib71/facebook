SELECT users.*, post.*, profile.* from users, post, profile WHERE (users.user_id = 30 AND post.postBy = 30 AND profile.userId = 30 AND post.post_id != (select post.post_id from post where post.postBy = 30 AND post.shareId != 'NULL') ) OR (users.user_id = 30 AND profile.userId =30 AND (post.sharedBy = 30 OR post.postBy =30 ) ) ORDER BY post.postedOn DESC LIMIT 10
